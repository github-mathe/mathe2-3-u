\Aufgabe[e]{Determinanten}{
Bestimmen Sie die Determinante der folgenden Matrix mit Hilfe des Gau\ss{}'schen
Algorithmus:
$$\boldsymbol B = 
	\begin{pmatrix}
	 1 &  2 &  1 & -3 &  0 \\
	-3 & -6 & -4 &  7 &  1 \\
	 2 &  5 &  3 & -5 &  2 \\
	-3 & -8 & -2 &  7 &  3 \\
	 0 &  2 &  1 &  6 &  3
	\end{pmatrix}
$$
}
\Loesung{
\begin{align*}
\operatorname{det} \boldsymbol B & = \operatorname{det}
	\begin{pmatrix}
	 1 &  2 &  1 & -3 &  0 \\
	 0 &  0 & -1 & -2 &  1 \\
	 0 &  1 &  1 &  1 &  2 \\
	 0 & -2 &  1 & -2 &  3 \\
	 0 &  2 &  1 &  6 &  3
	\end{pmatrix} \quad &(\text{tausche } Z_2 \text{ und } Z_3)\\
	& = -1\cdot\operatorname{det}
	\begin{pmatrix}
	 1 &  2 &  1 & -3 &  0 \\
	 0 &  1 &  1 &  1 &  2 \\
	 0 &  0 & -1 & -2 &  1 \\
	 0 & -2 &  1 & -2 &  3 \\
	 0 &  2 &  1 &  6 &  3
	\end{pmatrix}&(Z_4 + 2 \cdot Z_2;\; Z_5 - 2 \cdot Z_2)\\
	& = -1\cdot\operatorname{det}
	\begin{pmatrix}
	 1 &  2 &  1 & -3 &  0 \\
	 0 &  1 &  1 &  1 &  2 \\
	 0 &  0 & -1 & -2 &  1 \\
	 0 &  0 &  3 &  0 &  7 \\
	 0 &  0 & -1 &  4 & -1
	\end{pmatrix} &(Z_4 + 3 \cdot Z_3;\; Z_5 - Z_3)\\
	& = -1\cdot\operatorname{det}
	\begin{pmatrix}
	 1 &  2 &  1 & -3 &  0 \\
	 0 &  1 &  1 &  1 &  2 \\
	 0 &  0 & -1 & -2 &  1 \\
	 0 &  0 &  0 & -6 & 10 \\
	 0 &  0 &  0 &  6 & -2
	\end{pmatrix}&(Z_5 + Z_4)\\
	& = -1\cdot\operatorname{det}
	\begin{pmatrix}
	 1 &  2 &  1 & -3 &  0 \\
	 0 &  1 &  1 &  1 &  2 \\
	 0 &  0 & -1 & -2 &  1 \\
	 0 &  0 &  0 & -6 & 10 \\
	 0 &  0 &  0 &  0 &  8
	\end{pmatrix}\\ 
	& = (-1)\cdot 1\cdot 1\cdot(-1)\cdot(-6)\cdot 8 = -48\ .	
\end{align*}

}

\ErgebnisC{dummy001}
{
{a)} $\operatorname{det}\boldsymbol B = -48 $
