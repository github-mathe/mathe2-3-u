\end{align*}

$$B_\perp=\left\{\frac 1{\sqrt 6}\begin{pmatrix}1\\2\\-1\end{pmatrix},\, \frac 1{\sqrt 2}\begin{pmatrix}1\\0\\1 \end{pmatrix}\right\}.$$
