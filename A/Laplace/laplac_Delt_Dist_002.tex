\Aufgabe[e]{$\delta$--Distribution}
{
Bestimmen Sie die folgenden Integrale:
\begin{align*}
\mathbf{i)} &  & I_1 = \int\limits_{-\infty }^{\infty} \dfrac{\cos (x)}{1+x^2}\cdot \delta 
\left( x-\pi \right) \;\text{d}x\,. &  & 
\mathbf{ii)} &  & I_2 = \int\limits_{-\pi /2}^{\pi /2}\cos (x)\cdot \delta \left( x-\pi \right) 
\;\text{d}x\,.
\end{align*}
}
\Loesung{
\begin{align*}
  \textbf{i)} & \ & I_1=\int\limits_{-\infty }^{\infty} \dfrac{\cos (x)}{1+x^2}\cdot \delta \left( x-\pi \right) \;\text{d}x 
                      =& \dfrac{\cos (\pi)}{1+\pi^2} = \dfrac{-1}{1+\pi^2}\ .\\
  \textbf{ii)} & \ & I_2=\int\limits_{-\pi /2}^{\pi /2}\cos (x)\cdot \delta \left( x-\pi \right) \;\text{d}x =& 0 \quad \text{da}\quad 
                     \pi\notin[-\pi/2\,,\,\pi/2]\ .
\end{align*}
}


\ErgebnisC{laplacDeltDist002}{
i) $I_1 =\dfrac{-1}{1+\pi^2}$\,, \; ii) $I_2 =0$\,.
}
