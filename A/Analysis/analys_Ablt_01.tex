\Aufgabe[e]{Ableitungen}{
Man berechne die erste Ableitung der folgenden Funktionen
\begin{abc}
\item $f(x) = x^x$
\item $g(x) = x^{3^x}$
\item $h(x) = x^{\cos(x)}$
\end{abc}
}

\Loesung{

Wir nutzen die folgende Identität:
$$
a^b = \operatorname{e}^{b\ln(a)}.
$$

\begin{abc}
\item 
Wir schreiben zunächst:

$
f(x) = \operatorname{e}^{x\ln(x)}
$

Mit der Ketten- und der Produktregel erhalten wir die Ableitung

$
f'(x) = 
x^x\left(\ln\left(x\right)+1\right)
$

\item 
Wir schreiben $g(x)$ als:

$
g(x) = \operatorname{e}^{\operatorname{e}^{x\ln(3)}\ln(x)}
$

Mit der Ketten- und der Produktregel erhalten wir die Ableitung

$g'(x) = x^{3^x}\left(\ln\left(3\right)3^x\ln\left(x\right)+\dfrac{3^x}{x}\right)$

\item
Wir schreiben $h(x)$ als:

$
h(x) = \operatorname{e}^{\cos(x)\ln(x)}
$

Mit der Ketten- und Produktregel erhalten wir die Ableitung

$h'(x) = x^{\cos\left(x\right)}\left(\dfrac{\cos\left(x\right)}{x}-\ln\left(x\right)\sin\left(x\right)\right)$
\end{abc}

}


\ErgebnisC{analysAblt01}
{
\begin{itemize}
\item $f'(x) = x^x\left(\ln\left(x\right)+1\right)$
\item $g'(x) = x^{3^x}\left(\ln\left(3\right)3^x\ln\left(x\right)+\dfrac{3^x}{x}\right)$
\item $h'(x) = x^{\cos\left(x\right)}\left(\dfrac{\cos\left(x\right)}{x}-\ln\left(x\right)\sin\left(x\right)\right)$
\end{itemize}
}
