\Aufgabe[e]{Taylor--Entwicklung}{
Gegeben sei die Funktion
  \[
  f(x) := x\cdot \arctan(x) - \dfrac{1}{2}\ln(1+x^2)\,.
\]

\textbf{a)} \ Berechnen Sie das Taylorpolynom \ $T_2(x)$ \ zweiten Grades im Entwicklungspunkt \ $x_0=1$\,. 

\textbf{b)} \ Berechnen Sie das zugehörige Restglied und geben Sie den Fehler \ $|f(x)-T_2(x)|$ \ im Intervall \ $|x-1|\leq 0.1$ \ an.
}

\Loesung{
\textbf{a)} \ Es gilt
$$T_2(x) = f(1) + f'(1) \cdot (x-1) + \frac{1}{2} f''(1) \cdot (x-1)^2$$ 
mit folgenden Werten:
$$f(1) = \arctan(1) - \frac{1}{2} \ln(2) = \underline{\frac{\pi}{4} - \frac{1}{2} \ln(2)}\,,$$

$$f'(x) = \arctan(x) + \frac{x}{1+x^2} - \frac{1}{2} \frac{2x}{1+x^2} = \arctan(x) \Rightarrow \underline{f'(1)=\frac{\pi}{4}}\,,$$

$$f''(x) = \frac{1}{1+x^2} \Rightarrow \underline{f''(1) = \frac{1}{2}}\,,$$

$$f'''(x) = - \frac{2x}{(1+x^2)^2}\,.$$

Also gilt \ 
  \[
 T_2(x) = \left( \frac{\pi}{4} - \frac{1}{2} \ln 2\right) + \frac{\pi}{4}\cdot(x-1) + \frac{1}{4}\cdot(x-1)^2\,. 
\]


\textbf{b)} \ Für das Restglied gilt
$$
R_2(x;1)= f'''(\xi)\cdot\frac{(x-1)^3}{3!}  = - \frac{\xi}{\big(1+\xi^2\big)^2}\cdot\frac{(x-1)^3}{3} 
$$ 
mit \ $\xi = 1+ \theta(x-1)\,,\ \theta \in\, ]0,1[$\ . 

Für \ $|x-1| \leq 0.1$ \  gilt dann die Fehlerabschätzung
$$|f(x) - T_2(x)| =
  |R_2(x;1)| \leq \frac{1.1}{(1+0.9^2)^2} \cdot \frac{(0.1)^3}{3}  
\leq \frac{11}{(3/2)^2\cdot 3} \cdot 10^{-4} \leq 2\cdot 10^{-4}.
$$
}
